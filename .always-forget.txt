\begin{algorithm}[H]
\begin{algorithmic}[1]
        \Function{Find-Ind-Set}{$G = (V, E)$}
        \State{$S \gets \varnothing$}
        \For{$v\in V$}
        \State{Randomly sample a random real $r_v\in[0,1]$}
        \EndFor{}
        \State{$S \gets S \cup \set{v}$}
        \EndIf{}
        \EndFor{}
        \State{\Return{S}}
        \EndFunction{}
\end{algorithmic}
